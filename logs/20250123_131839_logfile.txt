    // ------------------------------
    //   COMMENT FUNCTIONS (LOCAL ONLY)
    // ------------------------------
    const comments = [];
    function addComment() {
      const name = document.getElementById('name').value.trim();
      const comment = document.getElementById('comment').value.trim();
      if (!name || !comment) {
        alert("Please provide both name and comment.");
        return;
      }
      const commentObject = { name, comment };
      comments.push(commentObject);
      displayComments();
      clearInputFields();
    }
    function displayComments() {
      const commentsContainer = document.getElementById('commentsContainer');
      commentsContainer.innerHTML = '';
      comments.forEach((comment, index) => {
        const commentDiv = document.createElement('div');
        commentDiv.className = 'comment';
        commentDiv.innerHTML = `
          <strong>${comment.name}</strong>
          <p>${comment.comment}</p>
          <button onclick="deleteComment(${index})">Delete</button>
        `;
        commentsContainer.appendChild(commentDiv);
      });
    }
    function deleteComment(index) {
      comments.splice(index, 1);
      displayComments();
    }
    function clearInputFields() {
      document.getElementById('name').value = '';
      document.getElementById('comment').value = '';
    }
    function goToLogPage() {
      sessionStorage.setItem('comments', JSON.stringify(comments));
      window.location.href = 'log.html';
    }

    // ------------------------------
    //   CHUNKED UPLOAD FUNCTION
    // ------------------------------
    const uploadEndpoint = "/upload-chunk"; // Adjust to your actual server endpoint
    const CHUNK_SIZE = 1024 * 1024; // 1 MB for example
    let offset = 0;
    let file;
    let totalSize;
    let fileName;

    function uploadFileChunked() {
      const fileInput = document.getElementById("fileInput");
      if (!fileInput.files.length) {
        alert("Please select a file before uploading.");
        return;
      }
      file = fileInput.files[0];
      totalSize = file.size;
      fileName = file.name;
      const statusDiv = document.getElementById("uploadStatus");
      statusDiv.innerHTML = `Preparing to upload "${file.name}"...`;
      offset = 0;
      sendChunks();
    }

    async function sendChunks() {
      const statusDiv = document.getElementById("uploadStatus");
      const formData = new FormData();
      const headers = new Headers();
      headers.append("fileName", fileName);
      headers.append("totalSize", totalSize);

      while (offset < totalSize) {
        const start = offset;
        const end = Math.min(start + CHUNK_SIZE, totalSize);
        const chunk = file.slice(start, end);
        if (chunk.size > 0) {
          formData.append("fileChunk", chunk, file.name);
        }
        offset += CHUNK_SIZE;
      }
      try {
        const response = await fetch(uploadEndpoint, {
          method: "POST",
          headers: headers,
          body: formData,
        });
        if (response.ok) {
          statusDiv.innerHTML = `Upload complete for "${file.name}"!`;
        } else {
          statusDiv.innerHTML = `Failed to upload: ${response.statusText}`;
        }
      } catch (error) {
        statusDiv.innerHTML = `Error uploading file: ${error.message}`;
      }
    }

    // ------------------------------
    //   CGI FUNCTIONS
    // ------------------------------
    async function callCalculatorCGI() {
      const input = document.getElementById('calcInput').value.trim();
      if (!input) {
        alert('Please provide input for the calculator.');
        return;
      }
      try {
        const response = await fetch('http://localhost:8080/cgi-bin/calculator.py', {
          method: 'POST',
          headers: {
            'Content-Type': 'text/plain' // Sending raw text as the body
          },
          body: input // e.g., "2+2"
        });
        if (!response.ok) {
          throw new Error(`Server returned status ${response.status}`);
        }
        const result = await response.text();
        alert(`Result: ${result}`);
      } catch (error) {
        alert(`Error: ${error.message}`);
      }
    }

    async function callHelloCGI() {
      const name = document.getElementById('helloInput').value.trim();
      if (!name) {
        alert('Please provide a name for the Hello CGI.');
        return;
      }
      try {
        const response = await fetch('http://localhost:8080/cgi/hello.php', {
          method: 'POST',
          headers: {
            'Content-Type': 'text/plain'
          },
          body: name
        });
        if (!response.ok) {
          throw new Error(`Server returned status ${response.status}`);
        }
        const text = await response.text();
        alert(text);
      } catch (error) {
        alert(`Error: ${error.message}`);
      }
    }

    // ------------------------------
    //   NEW: LIST & DELETE FILES
    // ------------------------------
    async function fetchUploads() {
      try {
        // We'll do a GET request to "/list-uploads" (adjust if needed)
        const response = await fetch('/list-uploads');
        if (!response.ok) {
          throw new Error(`Server error: ${response.status}`);
        }
        // Expect a JSON array of filenames
        const fileList = await response.json();

        const listDiv = document.getElementById('uploadList');
        listDiv.innerHTML = ''; // Clear previous
        if (fileList.length === 0) {
          listDiv.innerHTML = '<p>No uploaded files found.</p>';
        } else {
          // Show them in a list
          const ul = document.createElement('ul');
          fileList.forEach(file => {
            const li = document.createElement('li');
            li.textContent = file;
            ul.appendChild(li);
          });
          listDiv.appendChild(ul);
        }
      } catch (err) {
        alert('Failed to list uploads: ' + err.message);
      }
    }

    async function deleteFile() {
      const filename = document.getElementById('deleteFileName').value.trim();
      if (!filename) {
        alert("Please enter a file name to delete.");
        return;
      }
      try {
        // We'll do DELETE /uploads/<filename>
        const response = await fetch('/uploads/' + encodeURIComponent(filename), {
          method: 'DELETE'
        });

        }
        // const text = await response.text();

      } catch (err) {

    // ------------------------------
    //   COMMENT FUNCTIONS (LOCAL ONLY)
    // ------------------------------
    const comments = [];
    function addComment() {
      const name = document.getElementById('name').value.trim();
      const comment = document.getElementById('comment').value.trim();
      if (!name || !comment) {
        alert("Please provide both name and comment.");
        return;
      }
      const commentObject = { name, comment };
      comments.push(commentObject);
      displayComments();
      clearInputFields();
    }
    function displayComments() {
      const commentsContainer = document.getElementById('commentsContainer');
      commentsContainer.innerHTML = '';
      comments.forEach((comment, index) => {
        const commentDiv = document.createElement('div');
        commentDiv.className = 'comment';
        commentDiv.innerHTML = `
          <strong>${comment.name}</strong>
          <p>${comment.comment}</p>
          <button onclick="deleteComment(${index})">Delete</button>
        `;
        commentsContainer.appendChild(commentDiv);
      });
    }
    function deleteComment(index) {
      comments.splice(index, 1);
      displayComments();
    }
    function clearInputFields() {
      document.getElementById('name').value = '';
      document.getElementById('comment').value = '';
    }
    function goToLogPage() {
      sessionStorage.setItem('comments', JSON.stringify(comments));
      window.location.href = 'log.html';

    // ------------------------------
    //   CHUNKED UPLOAD FUNCTION
    // ------------------------------
    const uploadEndpoint = "/upload-chunk"; // Adjust to your actual server endpoint
    const CHUNK_SIZE = 1024 * 1024; // 1 MB for example
    let offset = 0;
    let file;
    let totalSize;
    let fileName;

    function uploadFileChunked() {
      const fileInput = document.getElementById("fileInput");
      if (!fileInput.files.length) {
        alert("Please select a file before uploading.");
        return;
      }
      file = fileInput.files[0];
      totalSize = file.size;
      fileName = file.name;
      const statusDiv = document.getElementById("uploadStatus");
      statusDiv.innerHTML = `Preparing to upload "${file.name}"...`;
      offset = 0;
      sendChunks();
    }

    async function sendChunks() {
      const statusDiv = document.getElementById("uploadStatus");
      const formData = new FormData();
      const headers = new Headers();
      headers.append("fileName", fileName);
      headers.append("totalSize", totalSize);

      while (offset < totalSize) {
        const start = offset;
        const end = Math.min(start + CHUNK_SIZE, totalSize);
        const chunk = file.slice(start, end);
        if (chunk.size > 0) {
          formData.append("fileChunk", chunk, file.name);
        }
        offset += CHUNK_SIZE;
      }
      try {
        const response = await fetch(uploadEndpoint, {
          method: "POST",
          headers: headers,
          body: formData,
        });
        if (response.ok) {
          statusDiv.innerHTML = `Upload complete for "${file.name}"!`;
        } else {
          statusDiv.innerHTML = `Failed to upload: ${response.statusText}`;
        }
      } catch (error) {
        statusDiv.innerHTML = `Error uploading file: ${error.message}`;
      }
    }

    // ------------------------------
    //   CGI FUNCTIONS
    // ------------------------------
    async function callCalculatorCGI() {
      const input = document.getElementById('calcInput').value.trim();
      if (!input) {
        alert('Please provide input for the calculator.');
        return;
      }
      try {
        const response = await fetch('http://localhost:8080/cgi-bin/calculator.py', {
          method: 'POST',
          headers: {
            'Content-Type': 'text/plain' // Sending raw text as the body
          },
          body: input // e.g., "2+2"
        });
        if (!response.ok) {
          throw new Error(`Server returned status ${response.status}`);
        }
        const result = await response.text();
        alert(`Result: ${result}`);
      } catch (error) {
        alert(`Error: ${error.message}`);
      }
    }

    async function callHelloCGI() {
      const name = document.getElementById('helloInput').value.trim();
      if (!name) {
        alert('Please provide a name for the Hello CGI.');
        return;
      }
      try {
        const response = await fetch('http://localhost:8080/cgi/hello.php', {
          method: 'POST',
          headers: {

        alert(text);
      } catch (error) {
        alert(`Error: ${error.message}`);
      }
    }

    // ------------------------------
    //   NEW: LIST & DELETE FILES
    // ------------------------------
    async function fetchUploads() {
      try {
        // We'll do a GET request to "/list-uploads" (adjust if needed)
        const response = await fetch('/list-uploads');
        if (!response.ok) {
          throw new Error(`Server error: ${response.status}`);
        }
        // Expect a JSON array of filenames
        const fileList = await response.json();

        const listDiv = document.getElementById('uploadList');
        listDiv.innerHTML = ''; // Clear previous
        if (fileList.length === 0) {
          listDiv.innerHTML = '<p>No uploaded files found.</p>';
        } else {
          // Show them in a list
          const ul = document.createElement('ul');
          fileList.forEach(file => {
            const li = document.createElement('li');
            li.textContent = file;
            ul.appendChild(li);
          });
          listDiv.appendChild(ul);
        }
      } catch (err) {
        alert('Failed to list uploads: ' + err.message);
      }
    }

    async function deleteFile() {
      const filename = document.getElementById('deleteFileName').value.trim();
      if (!filename) {
        alert("Please enter a file name to delete.");
        return;
      }
      try {
        // We'll do DELETE /uploads/<filename>

    const comments = [];
    function addComment() {
      const name = document.getElementById('name').value.trim();
      const comment = document.getElementById('comment').value.trim();
      if (!name || !comment) {

        return;
      }
      const commentObject = { name, comment };
      comments.push(commentObject);
      displayComments();
      clearInputFields();
    }
    function displayComments() {
      const commentsContainer = document.getElementById('commentsContainer');
      commentsContainer.innerHTML = '';
      comments.forEach((comment, index) => {
        const commentDiv = document.createElement('div');
        commentDiv.className = 'comment';
        commentDiv.innerHTML = `

        commentsContainer.appendChild(commentDiv);
      });
    }
    function deleteComment(index) {
      comments.splice(index, 1);
      displayComments();
    }

    // ------------------------------
    //   CHUNKED UPLOAD FUNCTION
    // ------------------------------
    const uploadEndpoint = "/upload-chunk"; // Adjust to your actual server endpoint
    const CHUNK_SIZE = 1024 * 1024; // 1 MB for example
    let offset = 0;
    let file;
    let totalSize;
    let fileName;

    function uploadFileChunked() {
      const fileInput = document.getElementById("fileInput");
      if (!fileInput.files.length) {
        alert("Please select a file before uploading.");
        return;
      }
      file = fileInput.files[0];
      totalSize = file.size;
      fileName = file.name;
      const statusDiv = document.getElementById("uploadStatus");
      statusDiv.innerHTML = `Preparing to upload "${file.name}"...`;
      offset = 0;
      sendChunks();
    }

    async function sendChunks() {
      const statusDiv = document.getElementById("uploadStatus");
      const formData = new FormData();
      const headers = new Headers();
      headers.append("fileName", fileName);
      headers.append("totalSize", totalSize);

      while (offset < totalSize) {
        const start = offset;
        const end = Math.min(start + CHUNK_SIZE, totalSize);
        const chunk = file.slice(start, end);
        if (chunk.size > 0) {
          formData.append("fileChunk", chunk, file.name);
        }
        offset += CHUNK_SIZE;
      }
      try {
        const response = await fetch(uploadEndpoint, {
          method: "POST",
          headers: headers,
          body: formData,
        });
        if (response.ok) {
          statusDiv.innerHTML = `Upload complete for "${file.name}"!`;
        } else {
          statusDiv.innerHTML = `Failed to upload: ${response.statusText}`;
        }
      } catch (error) {
        statusDiv.innerHTML = `Error uploading file: ${error.message}`;
      }
    }

    async function callCalculatorCGI() {
      const input = document.getElementById('calcInput').value.trim();
      if (!input) {
        alert('Please provide input for the calculator.');
        return;
      }
      try {
        const response = await fetch('http://localhost:8080/cgi-bin/calculator.py', {
          method: 'POST',
          headers: {
            'Content-Type': 'text/plain' // Sending raw text as the body
          },
          body: input // e.g., "2+2"
        });
        if (!response.ok) {
          throw new Error(`Server returned status ${response.status}`);
        }
        const result = await response.text();
        alert(`Result: ${result}`);
      } catch (error) {
        alert(`Error: ${error.message}`);
      }
    }

    async function callHelloCGI() {
      const name = document.getElementById('helloInput').value.trim();
      if (!name) {
        alert('Please provide a name for the Hello CGI.');
        return;
      }
      try {
        const response = await fetch('http://localhost:8080/cgi/hello.php', {
          method: 'POST',
          headers: {
            'Content-Type': 'text/plain'
          },
          body: name
        });
        if (!response.ok) {
          throw new Error(`Server returned status ${response.status}`);
        }
        const text = await response.text();
        alert(text);
      } catch (error) {
        alert(`Error: ${error.message}`);
      }
    }

    async function fetchUploads() {

        const response = await fetch('/list-uploads');

    // ------------------------------
    //   CHUNKED UPLOAD FUNCTION
    // ------------------------------
    const uploadEndpoint = "/upload-chunk"; // Adjust to your actual server endpoint
    const CHUNK_SIZE = 1024 * 1024; // 1 MB for example
    let offset = 0;
    let file;
    let totalSize;
    let fileName;

    function uploadFileChunked() {
      const fileInput = document.getElementById("fileInput");
      if (!fileInput.files.length) {
        alert("Please select a file before uploading.");
        return;
      }
      file = fileInput.files[0];
      totalSize = file.size;
      fileName = file.name;
      const statusDiv = document.getElementById("uploadStatus");
      statusDiv.innerHTML = `Preparing to upload "${file.name}"...`;
      offset = 0;
      sendChunks();
    }

    async function sendChunks() {
      const statusDiv = document.getElementById("uploadStatus");
      const formData = new FormData();
      const headers = new Headers();
      headers.append("fileName", fileName);
      headers.append("totalSize", totalSize);

      while (offset < totalSize) {
        const start = offset;
        const end = Math.min(start + CHUNK_SIZE, totalSize);
        const chunk = file.slice(start, end);
        if (chunk.size > 0) {
          formData.append("fileChunk", chunk, file.name);
        }
        offset += CHUNK_SIZE;
      }
      try {
        const response = await fetch(uploadEndpoint, {
          method: "POST",
          headers: headers,
          body: formData,
        });
        if (response.ok) {
          statusDiv.innerHTML = `Upload complete for "${file.name}"!`;
        } else {
          statusDiv.innerHTML = `Failed to upload: ${response.statusText}`;
        }
      } catch (error) {
        statusDiv.innerHTML = `Error uploading file: ${error.message}`;
      }
    }
