    const comments = [];
    function addComment() {
      const name = document.getElementById('name').value.trim();
      const comment = document.getElementById('comment').value.trim();

    function displayComments() {

    //   CHUNKED UPLOAD FUNCTION
    // ------------------------------
    const uploadEndpoint = "/upload-chunk"; // Adjust this to your actual server endpoint
    const CHUNK_SIZE = 1024 * 256; // 512 KB for demo
    let offset = 0;
    let file;
    let totalSize;
    let fileName;
    function uploadFileChunked() {
      const fileInput = document.getElementById("fileInput");
      if (!fileInput.files.length) {
        alert("Please select a file before uploading.");
        return;
      }
      file = fileInput.files[0];
      totalSize = file.size;
      fileName = file.name;
      const statusDiv = document.getElementById("uploadStatus");
      statusDiv.innerHTML = `Preparing to upload "${file.name}"...`;
      offset = 0; // Reset offset for each new upload
      // Send a single POST request with all chunks
      sendChunks();
    }
    // Function to send all chunks in one request
    async function sendChunks() {
      const statusDiv = document.getElementById("uploadStatus");
      const formData = new FormData();
      const headers = new Headers();
      // Add metadata to headers
      headers.append("fileName", fileName);
      headers.append("totalSize", totalSize);
      // Loop through file and append chunks to FormData
      while (offset < totalSize) {
        const start = offset;
        const end = Math.min(start + CHUNK_SIZE, totalSize);
        const chunk = file.slice(start, end);
        if (chunk.size > 0) { // Only append file chunks that contain data
          formData.append("fileChunk", chunk, file.name);
        }
        offset += CHUNK_SIZE; // Update offset for the next chunk
      }
      // Send the entire file and its chunks in a single POST request
      try {
        const response = await fetch(uploadEndpoint, {
          method: "POST",
          headers: headers, // Metadata in headers
          body: formData,   // All file chunks in the body
        });
        if (response.ok) {
          statusDiv.innerHTML = `Upload complete for "${file.name}"!`;
        } else {
          statusDiv.innerHTML = `Failed to upload: ${response.statusText}`;
        }
      } catch (error) {
        statusDiv.innerHTML = `Error uploading file: ${error.message}`;
      }
    }

    async function callCalculatorCGI() {
      const input = document.getElementById('calcInput').value.trim();
      if (!input) {
        alert('Please provide input for the calculator.');
        return;
      }
      const [num1, op, num2] = parseCalculationInput(input);
      if (!num1 || !op || !num2) {
        alert('Invalid input format. Use the format: num1 op num2 (e.g., 2+2).');
        return;
      }
      try {
        const response = await fetch(`http://localhost:8080/cgi-bin/calculator.py`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
          body: `num1=${num1}&num2=${num2}&op=${encodeURIComponent(op)}\r\n`
        });
        const text = await response.text();
        document.getElementById('uploadStatus').innerHTML = text;
      } catch (error) {
        document.getElementById('uploadStatus').innerHTML = `Error: ${error.message}`;
      }
    }
    function parseCalculationInput(input) {
      const match = input.match(/(\d+)([+\-*/])(\d+)/);
      return match ? [match[1], match[2], match[3]] : [];
    }

          body: `name=${encodeURIComponent(name)}\r\n`
        });
        const text = await response.text();
        document.getElementById('uploadStatus').innerHTML = text;
      } catch (error) {
        document.getElementById('uploadStatus').innerHTML = `Error: ${error.message}`;
      }
    }
